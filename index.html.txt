                        <span class="font-semibold text-slate-600">${cardIndex + 1} / ${vocabData.length}</span>
                        <button onclick="nextCard()" class="px-4 py-2 bg-indigo-600 text-white font-bold rounded-lg ${cardIndex === vocabData.length - 1 ? 'opacity-50 pointer-events-none' : ''}">다음</button>
                    </div>
                `;
                
                container.innerHTML = `
                    <div class="text-center text-xs text-slate-400 mb-2">💡 카드를 터치하면 뜻과 어원이 나타납니다.</div>
                    <div class="card-flip w-full h-80 cursor-pointer" onclick="flipCard()">
                        <div id="card-inner" class="card-inner w-full h-full relative rounded-2xl shadow-lg border border-slate-200">
                            <div class="card-front absolute w-full h-full bg-indigo-50 flex flex-col justify-center items-center rounded-2xl p-6">
                                <span class="text-xs text-indigo-400 font-bold tracking-widest mb-4">WORD</span>
                                <h2 class="text-3xl font-extrabold text-indigo-700 tracking-tight">${item.word}</h2>
                            </div>
                            <div class="card-back absolute w-full h-full bg-white flex flex-col justify-between rounded-2xl p-6 overflow-y-auto">
                                <div>
                                    <span class="text-xs text-emerald-500 font-bold tracking-widest block mb-2">MEANING</span>
                                    <h3 class="text-xl font-bold text-slate-800 border-b pb-2 border-slate-100">${item.meaning}</h3>
                                </div>
                                <div class="my-3 text-left space-y-2">
                                    <p class="text-xs text-slate-600"><strong class="text-amber-600 block mb-0.5">💡 어원</strong> ${item.etymology}</p>
                                    <p class="text-xs text-slate-600"><strong class="text-emerald-600 block mb-0.5">🌿 동계어</strong> ${item.cognates}</p>
                                </div>
                                <span class="text-[10px] text-slate-300">다시 터치하면 앞면으로 돌아갑니다.</span>
                            </div>
                        </div>
                    </div>
                `;
            }
        }

        function flipCard() {
            document.getElementById('card-inner').classList.toggle('card-flipped');
        }

        function nextCard() {
            if (cardIndex < vocabData.length - 1) {
                cardIndex++;
                renderContent();
            }
        }

        function prevCard() {
            if (cardIndex > 0) {
                cardIndex--;
                renderContent();
            }
        }

        // Initialize App
        renderContent();
